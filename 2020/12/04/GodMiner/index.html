#include "iostream"
#include <cstdio>
using namespace std;

int n;
int cnt = 0;

void miner(int s, int left, int right, int up, int down){
    if(s == 1) return;
    if(s == 2) {
        printf("%d\n", up);
        printf("%d\n", right);
        printf("cnt=%d\n", cnt++);
        return;
    }
    int tmp = (s+1)/2;
    if(s%2 != 0){
        miner(tmp, left, left+tmp-1, up, down-tmp+1);
        miner(tmp, right-tmp+1, right, up+tmp-1, down);
        //右上
        for(int i=0; i<s-tmp; i++){
            printf("%d ", up+i);
        }
        printf("\n");
        for(int i=0; i<s-tmp; i++){
            printf("%d ", right-tmp+2+i);
        }
        printf("\n");
        printf("cnt=%d\n", cnt++);
        //左下
        for(int i=0; i<s-tmp; i++){
            printf("%d ", up+tmp+i);
        }
        printf("\n");
        for(int i=0; i<s-tmp; i++){
            printf("%d ", left+i);
        }
        printf("\n");
        printf("cnt=%d\n", cnt++);
    }
    if(s%2 == 0){
        miner(tmp+1, left, left+tmp, up, up+tmp);
        miner(tmp, right-tmp+1, right, down-tmp+1, down);
        //右上
        for(int i=0; i<tmp; i++){
            printf("%d ", up+i);
        }
        printf("\n");
        for(int i=0; i<tmp-1; i++){
            printf("%d ", left+tmp+1+i);
        }
        printf("\n");
        printf("cnt=%d\n", cnt++);
        //左下
        for(int i=0; i<tmp-1; i++){
            printf("%d ", up+tmp+1+i);
        }
        printf("\n");
        for(int i=0; i<tmp; i++){
            printf("%d ", left+i);
        }
        printf("\n");
        printf("cnt=%d\n", cnt++);
    }
}

int main(){
    cin >> n;
    miner(n, 1, n, 1, n);
    return 0;
}
#include "iostream"
#include <cstdio>
#include <vector>

using namespace std;

int n;
vector<vector<int>> solutionX(50, vector<int>());
vector<vector<int>> solutionY(50, vector<int>());
int check;


void mineSq(int left, int right, int up, int down){
    for(int i=up; i<= down; i++){
        printf("%d ", i);
    }
    printf("\n");
    for(int i=left; i<= right; i++){
        printf("%d ", i);
    }
    printf("\n");
}


void miner(int s, int left, int right, int up, int down, int opr){
    if(s == 3){
        solutionX[opr].push_back(up);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(right);
        opr++;
        solutionX[opr].push_back(up+1);
        solutionY[opr].push_back(right);
        opr++;
        solutionX[opr].push_back(up+2);
        solutionY[opr].push_back(left);
        opr++;
        check = opr;
        return;
    }
    if(s == 4){
        solutionX[opr].push_back(up);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        opr++;
        solutionX[opr].push_back(up+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        opr++;
        solutionX[opr].push_back(up+2);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+3);
        opr++;
        solutionX[opr].push_back(up+3);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        opr++;
        check = opr;
        return;
    }
    if(s == 5){
        solutionX[opr].push_back(up);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        opr++;
        solutionX[opr].push_back(up+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        opr++;
        solutionX[opr].push_back(up+2);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        opr++;
        solutionX[opr].push_back(up+3);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+4);
        opr++;
        solutionX[opr].push_back(up+4);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        opr++;
        check = opr;
        return;
    }
    if(s == 6){
        solutionX[opr].push_back(up);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        solutionY[opr].push_back(left+5);
        opr++;
        solutionX[opr].push_back(up+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        solutionY[opr].push_back(left+5);
        opr++;
        solutionX[opr].push_back(up+2);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+3);
        solutionY[opr].push_back(left+4);
        solutionY[opr].push_back(left+5);
        opr++;
        solutionX[opr].push_back(up+3);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+4);
        solutionY[opr].push_back(left+5);
        opr++;
        solutionX[opr].push_back(up+4);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+5);
        opr++;
        solutionX[opr].push_back(up+5);
        solutionY[opr].push_back(left);
        solutionY[opr].push_back(left+1);
        solutionY[opr].push_back(left+2);
        solutionY[opr].push_back(left+3);
        opr++;
        check = opr;
        return;
    }
    if(s%2 == 1){
        int mid = (s+1)/2;
        for(int i=left+mid; i<=right; i++){
            solutionY[opr].push_back(i);
        }
        for(int i=up; i<=up+mid-2; i++){
            solutionX[opr].push_back(i);
        }
        opr++;
        for(int i=left; i<=left+mid-2; i++){
            solutionY[opr].push_back(i);
        }
        for(int i=up+mid; i<=down; i++){
            solutionX[opr].push_back(i);
        }
        opr++;
        solutionX[opr].push_back(up+mid-1);
        for(int i=left; i<=right; i++){
            if(i==left+mid-2 || i==left+mid-1) continue;
            solutionY[opr].push_back(i);
        }
        opr++;
        solutionY[opr].push_back(left+mid-1);
        for(int i=up; i<=down; i++){
            if(i==up+mid-1 || i==up+mid) continue;
            solutionX[opr].push_back(i);
        }
        opr++;
        miner(mid-1, left, left+mid-2, up, up+mid-2, opr);
        miner(mid-1, left+mid, right, up+mid, down, opr);
        return;
    }
    if(s%2 == 0){
        int mid = s/2;
        for(int i=left+mid-1; i<=right; i++){
            solutionY[opr].push_back(i);
        }
        for(int i=up; i<=up+mid-2; i++){
            solutionX[opr].push_back(i);
        }
        opr++;
        for(int i=left; i<=left+mid; i++){
            solutionY[opr].push_back(i);
        }
        for(int i=up+mid+2; i<=down; i++){
            solutionX[opr].push_back(i);
        }
        opr++;
        solutionX[opr].push_back(up+mid-1);
        for(int i=left; i<=right; i++){
            if(i==left+mid-2 || i==left+mid-1) continue;
            solutionY[opr].push_back(i);
        }
        opr++;
        solutionX[opr].push_back(up+mid);
        for(int i=left; i<=right; i++){
            if(i==left+mid || i==left+mid-1) continue;
            solutionY[opr].push_back(i);
        }
        opr++;
        solutionX[opr].push_back(up+mid+1);
        for(int i=left; i<=left+mid-1; i++){
            solutionY[opr].push_back(i);
        }
        opr++;
        miner(mid-1, left, left+mid-2, up, up+mid-2, opr);
        miner(mid-1, left+mid+1, right, up+mid+1, down, opr);
        return;
    }
}


int main(){
    cin >> n;
    miner(n, 1, n, 1, n, 0);
    //cout << check << endl;

        for(int j=0; j<solutionX[i].size(); j++){

        printf("\n");
        for(int k=0; k<solutionY[i].size(); k++){
            printf("%d ", solutionY[i][k]);
        }
        printf("\n");
    }
    return 0;
}
