字典序排序

这道题乍一看都能想到在考排序，然而字符串的排序和一般意义上的数字排序不同。如果是大一上没有经常练习C语言并且还有些Naive的小白同学，如何进行字典序可能就需要一些时间思考。当你解决了字典序——【逐个比较字母表顺序，若都相等比较长度】，还会遇到一个事情：字符串数组好像不能进行常规意义上的交换和赋值TAT

char s[100][100];

//wrong!!!!
void swap(char *s1, char *s2, int j){
    char *tmp = s[j];
    s[j] = s[j+1];
    s[j+1] = s[j];
}

那怎么办呢，我们应该学过交换指针。我们可以额外再开一个数组char *strArr[100];保存的是指向相应字符串的指针。一旦需要交换两个字符串，交换strArr中的指针即可。

最后在进行输出的时候也是取相应的指针（原字符串数组的字符串之间的相对位置没有改变）

简单解释一下代码——基本模拟了题意；在进行字典序比较的时候，首先一一对应比较字母表序，若相等则比较长度，一旦出现逆序（前大于后）则交换；因为没有复杂度的要求，我再次使用了最简单的冒泡排序。

#include <stdio.h>

int strlen(char *s) {
    int len = 0;
    while (*(s + len) != '\0') {
        len++;
    }
    return len;
}


int main() {
    int T, N;
    scanf("%d", &T);

    for(int t=0; t<T; t++){
        scanf("%d", &N);
        char *strArr[100];
        char str[100][50];
        for(int i=0; i<N; i++){
            scanf("%s", str[i]);
            strArr[i] = str[i];
        }
        int flg;
        for(int i=0; i<N-1; i++){
            for(int j=0; j<N-i-1; j++, flg = 0){
                int len1 = strlen(strArr[j]);
                int len2 = strlen(strArr[j+1]);
                int test = (len1 < len2)? len1 : len2;
                for(int k=0; k<test; k++){
                    char s1 = strArr[j][k];
                    char s2 = strArr[j+1][k];
                    if(s2 < s1){
                        flg = 1;
                        break;
                    }
                }
                if(len2 > len1 && flg == 0){
                    flg = 1;
                }
                //交换字符串的指针；
                if(flg == 1){
                    char *tmp = strArr[j];
                    strArr[j] = strArr[j+1];
                    strArr[j+1] = tmp;
                }
            }
        }
        printf("%s\n%s", strArr[0], strArr[N-1]);
    }
}

模拟Split函数

字符串切割，也是split函数的模拟；考察的是字符串的模式匹配。同样，这里并没有考察算法，因此我们不需要过分考虑复杂度的问题。

这样以来，问题就没有那么复杂了。这里提到两种解法：Brute-Force解法（暴力解法）和KMP解法（不做过多解释，仅提供解法(._.)）

Brute-Force

这里主要复杂在得到结果之后的处理（x

我们首先得到模式串Target和待查找串Str以及他们的长度。对Str的每一个元素进行遍历，如下：

如果有匹配的，则记下它此时在数组中的index，然后再进行遍历输出。记得在两个模式串之间为空时，要输出”EmptyString”。

#include <stdio.h>

int strlen(char *s) {
    int len = 0;
    while (*(s + len) != '\0') {
        len++;
    }
    return len;
}

int main() {
    char str[100], target[100];

    scanf("%s", str);
    scanf("%s", target);

    int lenStr = strlen(str);
    int lenTarget = strlen(target);
    int last[100];

    for(int i=0; i<100; i++){
        last[i] = -1;
    }

    if (lenTarget > lenStr) {
        printf("%s\n", str);
    } else {
        int gap = lenStr - lenTarget;
        for (int i = 0; i <= gap; i++) {
            for (int j = 0; j < lenTarget; j++) {
                if (str[i + j] != target[j]) {
                    break;
                }
                if (j == lenTarget - 1) {
                    last[i] = i+j;
                }
            }
        }
    }

    for(int i=0; i<lenStr; i++){
        if(last[i]==-1){
            printf("%c", *(str+i));
        } else if (last[i]+lenTarget == last[lenTarget+i]){
            i = last[i];
            printf("\n");
            printf("EmptyString");
        } else {
            if(i != 0){
                i = last[i];
                printf("\n");
            }
        }
    }
    return 0;
}

#include <stdio.h>

next[100];
start[100];

int strlen(char *s) {
    int len = 0;
    while (*(s + len) != '\0') {
        len++;
    }
    return len;
}

void buildNext(int lenPattern, char *pattern){
    next[0] = 0;

    int i = 0;
    int x = 1;
    int now = 0;

    while (x < lenPattern){
        if(pattern[now] == pattern[x]){
            now++;
            x++;
            next[++i] = now;
        } else if (now != 0){
            now = next[now - 1];
        } else {
            next[++i] = 0;
            x++;
        }
    }
}

void search(char *str, char  *pattern){
    int tar = 0;
    int position = 0;

    while (tar < strlen(str)){
        if(str[tar] == pattern[position]){
            tar++;
            position++;
        } else if (position != 0){
            position = next[position - 1];
        } else {
            tar++;
        }

        if(position == strlen(pattern)){
            //printf("%d\n", tar-position+1);
            start[tar-position] = strlen(pattern);
            position = next[position-1];
        }
    }
}

int main() {
    char str[100];
    char target[100];

    scanf("%s%s", str, target);

    int lenStr = strlen(str);
    int lenTarget = strlen(target);

    for(int i=0; i<100; i++){
        start[i] = 0;  //初始化
    }

    buildNext(lenTarget, target);
    search(str, target);

    for(int i=0; i<lenStr; i++){
        if(start[i]!=0){
            if(i != 0) printf("\n");
            i += start[i]-1;
            if(start[i+1]!=0){
                printf("EmptyString");
            }
            continue;
        }
        printf("%c", str[i]);
    }
}
