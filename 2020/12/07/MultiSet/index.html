    for(int i=0; i<Q; i++){
        scanf("%d", &choice);
        switch (choice) {
            case 0:
                scanf("%d", &x);

int n;
ll mod;
ll lans=0;
int l[maxn], r[maxn], rnd[maxn], w[maxn];
ll  size[maxn], val[maxn];
set<ll>S;

//wæ˜¯æƒé‡ //læ˜¯å·¦å­èŠ‚ç‚¹çš„ç´¢å¼• //ræ˜¯å³å­ç»“ç‚¹çš„ç´¢å¼• //rndæ˜¯ä¼˜å…ˆçº§ //valæ˜¯å¯¹åº”çš„ç´¢å¼• //sizeæ˜¯æ ‘çš„èŠ‚ç‚¹æ€»æ•°->å‰ç¼€å’Œ
struct treap {
  int sz, rt;
  inline void pushup(int x) { size[x] = size[l[x]] + size[r[x]] + w[x]*val[x]; }
  void lrotate(int &k) {
    int t = r[k];
    r[k] = l[t];
    l[t] = k;
    size[t] = size[k];
    pushup(k);
    k = t;
  }

  void rrotate(int &k) {
    int t = l[k];
    l[k] = r[t];
    r[t] = k;
    size[t] = size[k];
    pushup(k);
    k = t;
  }

  void insert(int &k, ll x) {
    if (!k) {
      sz++;
      k = sz;
      size[k] = x;
      w[k] = 1;
      val[k] = x;
      rnd[k] = rand();
      return;
    }
    size[k]+=x;
    if (val[k] == x) {
      w[k]++;
    } else if (val[k] < x) {
      insert(r[k], x);
      if (rnd[r[k]] < rnd[k]) lrotate(k);
    } else {
      insert(l[k], x);
      if (rnd[l[k]] < rnd[k]) rrotate(k);
    }
  }

  void del(int &k, ll x) {
    if (!S.count(x))
        return;
    if (val[k] == x) {
      if (w[k] > 1) {
        w[k]--;
        size[k]-=x;
        return;
      }
      if (l[k] == 0 || r[k] == 0) {
          S.erase(x);
          k = l[k] + r[k];
      }
      else if (rnd[l[k]] < rnd[r[k]]) {
        rrotate(k);
        del(k, x);
      } else {
        lrotate(k);
        del(k, x);
      }
    } else if (val[k] < x) {
      size[k]-=x;
      del(r[k], x);
    } else {
      size[k]-=x;
      del(l[k], x);
    }
  }

  ll querySum(int k, ll x) {
    if (!k) return 0;
    if (val[k] == x)
      return size[l[k]] + w[k]*val[k];
    else if (x > val[k]) {
      return size[l[k]] + w[k]*val[k] + querySum(r[k], x);
    } else
      return querySum(l[k], x);
  }

  int querynum(int k, ll x) {
    if (!k) return 0;
    if (x == val[k]){
        return w[k];
    } else if (x < val[k]){
        return querynum(l[k], x);
    } else {
        return querynum(r[k], x);
    }
  }
} T;

}
