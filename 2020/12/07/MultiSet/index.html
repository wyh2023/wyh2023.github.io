MultiSet

题目

After learning binary search trees in algorithm class, Nezzar challenges you with the following task.

Given a multiset 𝑆S which is intially emtpy, Nezzar may do one of the following operations in each round, and your algorithm should support these features.

  • 0 $x$, insert $x$ to $S$;
  • 1 $x$, output the number of copies of $x$ in $S$, and remove exactly one copy of $x$ from $S$ if $S$ contains at least one copy of $x$;
  • 2 $l,r$, output $∑x∈S,l<x≤r$. For example, if $S={3,5,5,7},l=3,r=7$, then output should be $17$ as $5+5+7=17$.

To disallow some shortcut solutions (offline solutions), Nezzar will fix a positive integer 𝑚𝑜𝑑mod, and provide input in the following way (𝑙𝑎𝑛𝑠lans is 00 initially).

  • 0 $y$ where $0≤y<mod$, let $x=(y+lans)$ mod $mod$ and insert $x$ to $S$;
  • 1 $y$ where $0≤y<mod$, let $x=(y+lans)$ mod $mod$ and output the number of copies of $x$ in $S$, then remove exactly one copy of $x$ from $S$ if $S$ contains at least one copy of $x$;
  • 2 $u,v$ where $0≤u,v<mod$, let
    • $l=(u+lans)$ mod $mod$ and $r=(v+lans)$ mod $mod$, swap $l$ and $r$ if $l>r$, output $∑x∈S,l<x≤r$ and finally update $lans$ with output.

Input

The first line of input contains two integers $Q$ and $mod$, where $1≤Q≤5⋅10^5$ and $1≤mod≤10^13$.

Each of the following $Q$ lines contains an operation described above.

Output

For each type 1 and type 2 query, output one line with the required value.

Sample Input

7 15
1 2
0 3
0 6
2 0 7
0 5
2 4 5
2 5 4

Sample Output

0
9
14
0

Note

Initially, $S={}$.

Parameter for the first query is $x=(y+lans)$ mod $mod=2$. Hence, output for this query should be 0, and $S={}$ after first query.

Parameter for the second query is $x=(y+lans)$ mod $mod=3$. Hence, $S={3}$ after the second query.

Hint

Your algorithm should answer each query in $O(log⁡n)$ time to get accepted. If you have no idea how to debug, first write a naive algorithm (i.e., a straightforward but slow algorithm), then generate some small random test cases, finally compare the naive algorithm’s output with your final fast algorithm’s output.

Naive Solution

#include "cstdio"
#include "set"
#include "iterator"
#include "map"

#define MAX 2147483647

using namespace std;

int Q;
int mod;
int choice;
int x;
int u, v;
int l, r;
int lans=0;
set<int>S;
map<int, int> numMap;

void multiS(){
    scanf("%d %d", &Q, &mod);
    for(int i=0; i<Q; i++){
        scanf("%d", &choice);
        switch (choice) {
            case 0:
                scanf("%d", &x);
                if(x>=0&&x<mod){x = (x+lans)%mod;}
                if(S.count(x)==0) numMap.insert({x, 0});
                numMap[x]++;
                S.insert(x);
                break;
            case 1:
                scanf("%d", &x);
                if(x>=0&&x<mod){x = (x+lans)%mod;}
                if(numMap[x]>0){
                    printf("%d\n", numMap[x]);
                } else {
                    printf("0\n");
                }
                if(numMap[x]==1){
                    S.erase(x);
                }
                if(numMap[x]>0){
                    numMap[x]--;
                }
                break;
            case 2:
                scanf("%d %d", &u, &v);
                if(u>=0&&u<mod){l = (u+lans)%mod;}
                if(v>=0&&v<mod){r = (v+lans)%mod;}
                if(l>r){
                    int tmp = l;
                    l = r;
                    r = tmp;
                }
                int sum = 0;
                pair<set<int>::const_iterator,set<int>::const_iterator> pr;
                pr = S.equal_range(l);
                for(auto i=pr.second; i!=S.upper_bound(r); i++){
                    int t = numMap[*i];
                    sum += (*i)*t;
                }
                lans = sum;
                printf("%d\n", sum);
                break;
        }
    }
    printf("Hello!");
}

int main(){
    S.insert(MAX);
    multiS();
}

维护Treap解法

#include <algorithm>
#include <cstdio>
#include <iostream>
#include "set"

using namespace std;

#define maxn 1000005
#define ll long long
#define INF (1 << 30)

int n;
ll mod;
ll lans=0;
int l[maxn], r[maxn], rnd[maxn], w[maxn];
ll  size[maxn], val[maxn];
set<ll>S;

//w是权重 //l是左子节点的索引 //r是右子结点的索引 //rnd是优先级 //val是对应的索引 //size是树的节点总数->前缀和
struct treap {
  int sz, rt;
  inline void pushup(int x) { size[x] = size[l[x]] + size[r[x]] + w[x]*val[x]; }
  void lrotate(int &k) {
    int t = r[k];
    r[k] = l[t];
    l[t] = k;
    size[t] = size[k];
    pushup(k);
    k = t;
  }

  void rrotate(int &k) {
    int t = l[k];
    l[k] = r[t];
    r[t] = k;
    size[t] = size[k];
    pushup(k);
    k = t;
  }

  void insert(int &k, ll x) {
    if (!k) {
      sz++;
      k = sz;
      size[k] = x;
      w[k] = 1;
      val[k] = x;
      rnd[k] = rand();
      return;
    }
    size[k]+=x;
    if (val[k] == x) {
      w[k]++;
    } else if (val[k] < x) {
      insert(r[k], x);
      if (rnd[r[k]] < rnd[k]) lrotate(k);
    } else {
      insert(l[k], x);
      if (rnd[l[k]] < rnd[k]) rrotate(k);
    }
  }

  void del(int &k, ll x) {
    if (!S.count(x))
        return;
    if (val[k] == x) {
      if (w[k] > 1) {
        w[k]--;
        size[k]-=x;
        return;
      }
      if (l[k] == 0 || r[k] == 0) {
          S.erase(x);
          k = l[k] + r[k];
      }
      else if (rnd[l[k]] < rnd[r[k]]) {
        rrotate(k);
        del(k, x);
      } else {
        lrotate(k);
        del(k, x);
      }
    } else if (val[k] < x) {
      size[k]-=x;
      del(r[k], x);
    } else {
      size[k]-=x;
      del(l[k], x);
    }
  }

  ll querySum(int k, ll x) {
    if (!k) return 0;
    if (val[k] == x)
      return size[l[k]] + w[k]*val[k];
    else if (x > val[k]) {
      return size[l[k]] + w[k]*val[k] + querySum(r[k], x);
    } else
      return querySum(l[k], x);
  }

  int querynum(int k, ll x) {
    if (!k) return 0;
    if (x == val[k]){
        return w[k];
    } else if (x < val[k]){
        return querynum(l[k], x);
    } else {
        return querynum(r[k], x);
    }
  }
} T;

int main() {
    srand(123);
    scanf("%d", &n);
    scanf("%lld", &mod);
    int opt;
    ll x;
    ll v;
    ll l, r;
    for (int i = 1; i <= n; i++) {
        scanf("%d%lld", &opt, &x);
        if (opt == 0) {
            if(x>=0&&x<mod) x = (x + lans) % mod;
            S.insert(x);
            T.insert(T.rt, x);
        }
        else if (opt == 1){
            if(x>=0&&x<mod) x = (x + lans) % mod;
            printf("%d\n", T.querynum(T.rt, x));
            T.del(T.rt, x);
        }
        else if (opt == 2) {
            scanf("%lld", &v);
            if(x>=0&&x<mod) l = (x+lans)%mod;
            if(v>=0&&v<mod) r = (v+lans)%mod;
            if(l>r){
                ll tmp = l;
                l = r;
                r = tmp;
            }
            ll sum = T.querySum(T.rt, r) - T.querySum(T.rt, l);
            printf("%lld\n", sum);
            lans = sum;
        }
    }
    return 0;
}
